# video test src over rtp
gst-launch-1.0 -v videotestsrc ! video/x-raw,framerate=20/1 ! videoscale ! videoconvert ! x264enc tune=zerolatency bitrate=500 speed-preset=superfast ! rtph264pay config-interval=10 pt=96 ! udpsink host=127.0.0.1 port=5000

# webcam over rtp
gst-launch-1.0 -v v4l2src device="/dev/video0" ! videoscale ! videoconvert ! x264enc tune=zerolatency bitrate=500 speed-preset=superfast ! rtph264pay config-interval=10 pt=96 ! udpsink host=127.0.0.1 port=5000

# wabcam to video sink and over rtp
gst-launch-1.0 -v v4l2src device="/dev/video0" ! videoscale ! videoconvert ! tee name="t" t. ! queue ! x264enc tune=zerolatency bitrate=500 speed-preset=superfast ! rtph264pay config-interval=10 pt=96 ! udpsink host=127.0.0.1 port=5000 t. ! queue ! autovideosink

# wabcam to video sink and over rtp after encoding
gst-launch-1.0 -v v4l2src device="/dev/video0" ! videoscale ! videoconvert ! x264enc tune=zerolatency bitrate=500 speed-preset=superfast ! rtph264pay config-interval=10 pt=96 ! tee name="t" t. ! queue ! udpsink host=127.0.0.1 port=5000 t. ! queue ! rtph264depay ! decodebin ! videoconvert ! autovideosink

# receive h264 rtp stream
gst-launch-1.0 -v udpsrc port=5000 caps = "application/x-rtp, media=(string)video, clock-rate=(int)90000, encoding-name=(string)H264, payload=(int)96" ! rtph264depay ! decodebin ! videoconvert ! autovideosink
